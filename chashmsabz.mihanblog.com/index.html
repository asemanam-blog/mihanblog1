تنگ دلم

بِدون كه من  دوسِت دارم
بخاطر دلتنگیهات
یه حالی دارم اینروزا
خرابِ دوری نگات
وقتیكه نیستی سایِبون
توی نِگام یه مِهربون
ناراحتم بی نغمه هات
همیشه دلتنگِ صدات
به مهربونیهات قَسم
میدونی اهل باورم
بزار خودمونی بِگم
تنگِ دِلم زیاد بَرات
بیا با مهربونیمون
پروانه من گُلی بمون
حرفِ همو خوب بخونیم
شهدِ نگاه و  غنچه هات
ما اومدیم با همدیگه
با همدِلی یكی بشیم
زِ خوبی و یكرنگی مون
لبخندِ من رویِ لَبات
نَشیم اسیرِ وَسوَسه
خودخواهیها دیگه بَسه
بدونیم قدرِ لحظه رو
بهارِ من شكوفه هات

مولا حسین


حسین عزیزِ زهرا
حسین حسین جان مولا


حسین زِ عشق تو من
همیشه در تب و تابم
بدستِ مامَت زهرا
سیراب مِهرِ نایابم


زِ کربلایِ مِهرت
شور و نَوا بِه دِل نشیند
گلابِ چشمِ ما را
یه نینوایِ تو بِچیند


همهء وجودم تشنه
به قطرهء صفایِ جودت
وفایِ عباس درسم
فدایی ام بَهرِ وجودت


به قامتِ آن رعنا
خیره ام و زِ دِل غمینم
علی بِکُشتند اکبر
شکسته قلب با تو حَزینم


بِخُشکد این حلقِ من
سه شعبه ای رفته به حَنجُر
بِدیده اند دستانت
پارگیِ گلویِ اصغر


پاشیده ای خونِ دِل
به آسمان هدیه خدایت
من هَمرهم تا فَرجام
جانِ همه عالم فدایت


نشسته ماندم با هِق هِق
یادِ رُقیه کنجِ خرابه
راس پدر در آغوش
سپرده جان حالِ اِنابه


غمزده ام با زینب
با کاروانش همرهِ سجاد
به عشق مرگِ سُرخی
مولا حسین گویم بِه فریاد

مهر ثارالله


آخرِ عشق ست چون مجنون شدن
گر حسین لیلایِ دِل ممنون شدن
عاشقی در مکتبِ عشقِ حسین
اشک باران در غمش محزون شدن


پیروِ راهِ خدا در زندگی
مستمر حتی اگر دِل خون شدن
مهرِ ثارالله با دلها چه کرد
عاقبت خَیر در رَهَش مامون شدن


شیعه با نامِ حسین پاینده است
گر نباشد پیروش مدیون شدن
خونِ خویش فرزند و یارانش فَدا
 قدر نشناسان چنان مَلعون شدن


عزّتی گر ما را باشد تاکنون
 از سَرِ لُطفِ حسین در خون شدن
کربلا تا اربعین اش در عزاست
زینبی ماندن بجاست مَرهون شدن


 زنده مانیم گر بدانیم  حُرمَتَش
مُرده آنست با ستم همخون شدن
لَعن و نفرین بر یزید و آل او
نَنگ  باشد زِ عدو ممنون شدن

حس من

پشتِ نگاهم حسّ من
زِ دل عزیز میخونمت
تو باغچهء باورِ دل
همیشه گُل میدونمت
تویِ هوایی مه گونه
پُر میشم از شبنم تو
من بُِامیدم یه روزی
چشام بِشه همدمِ تو
من میدونم به یادمی
زِ دِل بتو وابسته ام
دِلتنگتَم عزیزِ من
به یك نگاه  نشسته ام
 دُعا بكن گُل بگیره
وجودِ من توی هوات
پُرِ شكوفه ها كنم
مَسیرِ تو  تا به سَرات
در كویِ احساسِ تو من
رَدّ حضورت بمونم
زِ بویِ  مهربونی ات
لُطفِ تو رو خوب بخونم
چی میشه وقتی كه میآی
دِلِ من آروم بگیره
دلشوره ها كنار بِره
دلگرمی سامون  بگیره
صَحن و سَرای تو میشه
یه وعده گاهِ حرفِ دِل
دلها جُدا میشه زِ غیر
میآم پیشت گرچه خِجِل
آه كه چقدر طَلاییه
لحظهِ دیدارِ چِشات
بریزه اشكِ شوقِ من
به عشقِ حرفها و نِگات
لَب وا كنی یه عالمه
دِل تو دِلی نمیمونه
اَسیرِ خالِ لبِ تو
دیده به غیر نمیخونه
بیا عزیز یه دنیا حرف
مونده برایِ گفتنم
بُغضِ شكسته ی گلو
خستگیهایِ این تَنَم
من میدونم وقتی بیآی
دوستامو با خود میآری
اونهاییكه پَر كشیدند
هنگامِ رزم و پایداری
میخوام با حِسّ گفتنم
قلبَمو دَرمون بكنی
برای لیلی بودنت
نَبضَمو مجنون بكنی
