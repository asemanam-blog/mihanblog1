به منظور روشن شدن جایگاه تفسیر قرآن به قرآن در قبال سایر روشهای تفسیری و مقدار دخالت سنّت در آن و كیفیت جمع بین مناهج تفسیر درایی و روایی و لزوم رعایت ترتیب آنها و حلّ برخی از تعارضهای احتمالی، مطالبی ارائه می‏شود:
تفسیر یعنی تبیین معنای کلمه یا کلام خردمندانه
یكم. معنای تفسیر، چنان‏كه در مبحث تفسیر به رأی بازگو شد، روشن نمودن معنای كلمه یا كلام كه بر اساس قانون محاوره و فرهنگ مفاهمه ایراد شد. قهراً لفظی كه معنای آن ضروری است، بی‏نیاز از تفسیر می‏باشد؛ چنان‏كه كلمه یا كلامی كه با تعمیه و الغاز ادا شده و از سنخ معمّا و الغاز باشدْ، بر محور فرهنگ حِوار و تفاهم ایراد نشده است و حكم ویژهٴ خود را دارد.
بنابراین، لفظِ مفرد یا جمله‏ای كه با تدبّر و تأمّل خردمندانه، مبادی تصوّری و تصدیقی روشنی پیدا می‏كند، نیازمند به تفسیر است و تحلیل مبادی مزبور و رسیدن به مقصود متكلّم و مدلول بسیط و مركب لفظ، همانا تفسیر كلمات نظری خواهد بود و اختصاصی به متون دینی مخصوصاً قرآن كریم ندارد؛ گرچه شرح خصوص قرآن، به فنّ تفسیر معروف شده است.
تفسیرپذیری قرآن کریم
دوم. قرآن كریم تفسیر پذیر است: 1. چنان‏كه تفسیر او ضروری و لازم می‏باشد. 2. امّا اینكه قرآن تفسیر پذیر است، برای آنكه از گزند تفریطِ بداهتْ، منزّه و از آسیبِ افراطِ تعمیه و الغاز مصون می‏باشد. نه آنقدر ساده و مبتذل است كه نیازی به تحلیل مبادی تصوری و تصدیقی نداشته باشد، نه آنچنان پیچیده و مبهم است كه نظر معمّا و لغز خارج از قانونِ مفاهمه و فرهنگ حِوار باشد و در دسترس تفسیر قرار نگیرد؛ بلكه قرآن كریم در عین نور بودنْ، شمسی بر معارف بلند و مطالب ژرف است كه او را تفسیر پذیر می‏نماید.
اما اینكه تفسیر او ضروری و لازم است و بدون تفسیرْ قابل ادراك عموم نخواهد بود، برای همان است كه ضمناً اشاره شد و راز آن را خود قرآن بازگو می‏كند.
قرآن كریم چون، هم برای خود اوصافی یاد می‏كند كه ضرورت تفسیر آن را به همراه دارد و هم علومی را طرح می‏نماید كه بدون تفسیرْ ادراك نمی‏شود؛ زیرا از طرفی، هم خود را به عنوان سخن رصین و گفتار وزین و پر مغز می‏ستاید: ﴿إنّا سنلقی علیك قولاً ثقیلاً﴾[1] و هم سلسلهٴ جبال را در پیشگاه هَیْمَنَه و سیطرهٴ خود خاضع و خاشع و متصدّع و متلاشی می‏داند: ﴿لو أنزلنا هذا القران علی جبلٍ لرأیته خاشعاً متصدّعاً من خشیة اللّه و تلك الأمثال نضربها للناس لعلّهم یتفكرون﴾[2] و هم همگان، اعم از پری و انسان را به مصاف و مبارزه فرامی‏خواند و عجز آنان را در این تحدّیِ نَفْس گیر و جامع اعلام می‏دارد؛ ﴿قل لئن اجتمعت الإنس و الجن علی أن یأتوا بمثل هذا القران لا یأتون بمثله و لو كان بعضهم لبعضٍ ظهیراً﴾.[3]
ظهور اطلاقی جملهٴ ﴿لا یأتون﴾، این است كه جامعهٴ مجتمع انس و جن، برای همیشهٴ تاریخ در این پیكارِ اعجاز آمیز فرسوده و ناتوان‏اند؛ چنان‏كه از جملهٴ ﴿فإن لم تفعلوا و لن تفعلوا فاتّقوا النار الّتی وقودها الناس و الحجارة أُعدّت للكافرین﴾[4]، عجز جاودانهٴ محاربانْ در این نبرد معلوم می‏شود.
از طرف دیگر، علوم و معارف ویژه‏ای در جهان‏بینی توحیدی، اسمای حسنای الهی، صفات علیای ازلی، قضا و قدر، جبر و تفویض و اختیار، تجرّد روح، عصمت فرشتگان، عصمت و طهارت انبیا و ائمّه اهل بیت(علیهم‏السلام)، امامت و رهبری نظام اسلامی، داوری پیرامون مكاتیب دیگر، شرح ره توشهٴ انبیاء سلف، دستور اولیای خلف و دهها مسائل عمیق حكمت نظری و عملی دیگر را ارائه می‏نماید كه بدون شرح و توضیح خردمندان، ادراك عمومی آن میسور نخواهد بود؛ لذا ضرورت تفسیر قرآن از دو لحاظ است: یكی آنكه كتابِ عمیقِ علمی و وزینِ نظری قطعاً بدون تفسیر ادراك نمی‏شود (از جهت علمی) و دیگر آنكه كتابِ هدایت اگر پیام او این باشد كه: ﴿إنّ هذا القران یهدی للّتی هی أقوم و یبشّر المؤمنین الذین یعملون الصالحات أنّ لهم أجراً كبیراً﴾[5]، به منظور اهتداء جامعهٴ بشری، چاره‏ای جز تبیین مفاهیم و تفسیر معانی آن نیست (از جهت عملی).
 ضرورت تفسیر قرآن برای متضلّعانِ علومِ متنوّع و فنونِ متعدد آشكارتر است؛ لذا جریان تفسیر قرآن از عصر نزول تا كنون به عنوان یك سنّت حسنه توسط حضرت رسول‏اكرم(صلی الله علیه و آله و سلم) و اهل‏بیت عصمت و طهارت(علیهم‏السلام) و نیز به وسیلهٴ صحابه، تابعانِ اصحاب، قدماء و متأخران از علمای دین، دارج و رایج بوده است. گرچه بعضی از گذشتگان از بِدار به تفسیر قرآن مجید تحرّز داشته و برخی از آن تحاشی می‏ورزیدند و گروهی نسبت به آن تحرّج می‏نمودند، لیكن همگی از تفسیر به مأثور بهره‏برداری می‏كردند، گرچه عده‏ای از اظهار رأی خودداری می‏كردند، كه شرح آن در فصل تفسیر به رأی بیان شد، و تا قرن پنجم هجری غیر از تفسیر روایی روش دیگری به عنوان تفسیر درایی و اجتهادی رواج نداشت، گرچه فی الجمله مطرح بوده است؛ مانند آنچه پیرامون اجتهاد اَدَبی و لغوی در آثار سلف مشهود می‏باشد.
روشن بودن قرآن و بصیر بودن مفسر، مهم ترین شرط تفسیر قرآن
سوم. تفسیر قرآن گرچه شرایط و آداب متعددی دارد؛ لیكن مهم‏ترین شرط محوری آن لزوم روشن بودن قرآن از یك سو تا قابل دیدن و فهمیدن مراجعه كننده باشد و بصیر و بینا بودن مُفَسِّرین از سوی دیگر تا لایق دیدن و فهمیدن مطالب آن باشند؛ زیرا گرچه مثلاً خورشیدِ تابان روشن است؛ لیكن اَعْمی، اَعْوَر، اَحْول، اَكْمَه یا اصلاً او را نمی‏بیند و یا او را آنچنان كه هست مشاهده نمی‏كند.
قرآن كریم گرچه نور است: ﴿یا أیها الناس قد جائكم برهان من ربّكم و أنزلنا إلیكم نوراً مبیناً﴾؛[6] ﴿فامنوا باللّه و رسوله و النور الذی أنزلنا و اللّه بما تعملون خبیر﴾؛[7] ﴿...و اتّبعوا النور الذی أنزل معه...﴾؛[8] ﴿...قد جائكم من اللّه نور و كتاب مبین﴾[9]، لیكن نوری است ثقیل و وزین؛ نه خفیف و سخیف: ﴿إنّا سنلقی علیك قولاً ثقیلاً﴾[10]؛ لذا برای دیدن چنین نور وزینی چاره‏ای جز تحصیل بَصَر حدید و بینش عمیق علمی نخواهد بود.
بلكه دربارهٴ قرآن كریم گفته‏اند: «القُرآن غریمٌ لا یُقْضی دَیْنُه و غریب لا یُؤدّی حقّه»؛ یعنی قرآن طلبكاری است كه هرگز وام او پرداخت نمی‏شود و غریبی است كه هیچ‏گاه حق او اداء نخواهد شد؛ زیرا اوج معارف او و عمق مطالب او فقط ممسوس دست اندیشهٴ تابناك معصومین(علیهم‏السلام) می‏باشد؛ ﴿إنّه لقران كریم ٭ فی كتاب مكنون ٭ لا یمسّه إلاّ المطهّرون﴾.[11] از این جهت، در تعریف مفهوم تفسیر قرآن، قید «به قدر طاقت بشری» اخذ شده است.
بنابراین، ضمن لزوم بصیر و بینا بودن مُفَسِّر، لازم است برای بهروری بیشتر از معارف قرآن كریم به «مطهّرون» كه همان اهل‏بیت عصمت و طهارت‏اند(علیهم‏السلام) رجوع نمود؛ البته قلمرو رجوع به سنّت معصومین(علیهم‏السلام) روشن خواهد شد.
انواع راه های ممکن برای رساندن پیام الهی
چهارم. مقتضای برهان نبوّت عام كه شامل نبوت حضرت خاتم(صلی الله علیه و آله و سلم) نیز می‏شود، این است كه كتاب آسمانی و قانون الهی پیامبر، هم قابل فهم باشد و هم صالح برای عمل. آن كتابی كه قوانین آن مفهوم انسانها و یا قابل اجراء در جامعه نیست، شایستهٴ پیامبر خدا نخواهد بود؛ لیكن صلاحیت قانون پیامبر برای علم و عمل، منحصر در این نیست كه خودِ كتاب آسمانیْ تمام معارف و احكام را به طور تفصیل بیان نماید؛ بلكه ممكن است برخی از آنها را به طور تفصیل روشن كند و تفصیل بعض دیگر را از راه وحی و الهام به خود پیامبر برای جامعه تبیین نماید یا آنكه اصلاً به بیان خطوطِ كلیِ حِكَم و اَحْكام بسنده كند و تفصیل همهٴ آنها از راه حدیث قدسی به پیامبر الهام شود و آن حضرت(صلی الله علیه و آله و سلم) تفصیل دریافت شده را به امّت خویش ابلاغ نماید.
غرض آنكه آنچه از برهان ضرورت وحی و نبوّت استفاده می‏شود، همانا رسیدن پیام خدا در اصول دین و فروع آن به مردم است و آن بلاغ راههای متعددی دارد كه به رئوس آن اشارت رفت و هرگز منحصر در بیان تفصیلیِ خودِ متنِ مقدّسِ آسمانی نیست.
برای فهم متنِ مقدّس دینی، بهترین راه تدبّر تام در همان متنِ منزّه الهی می‏باشد، قرآن كریم نیز از آن جهت كه نور است، هیچ ابهام، تیرگی و تاریكی در آن نیست؛ گرچه وزین و عمیق است و وزانتِ علمی و عمق نظری او با نورانی بودن آن منافی نمی‏باشد؛ زیرا نور در قبال ظلمتِ ابهام است؛ نه در برابر نظری و ژَرف بودن تا اینكه نور بودن قرآن به معنای بدیهی بودن آن بوده و نیازی به تفسیر نداشته باشد.
گروهی كه در اثر رسوبِ مرسوم جاهلیت از یك سو و غبار اغیار دگراندیشی از سوی دیگر، گرفتار دو مانع علمی و عملی برای دیدن و فهمیدن نور وحی شده‏اند: ﴿كلاّ بل ران علی قلوبهم ما كانوا یكسبون﴾؛[12] ﴿...فإنّها لا تعمی الأبصار و لكن تعمی القلوب الّتی فی الصدور﴾[13]؛ ﴿و ما أنت بهادی العمی عن ضلالتهم...﴾[14] ، چون بصیر نیستند، از دیدن علمی نور وحیْ محجوب و از ادراك ضیاء الهام الهی محروم‏اند؛ ﴿...و طبع علی قلوبهم فهم لا یفقهون﴾[15]؛ ﴿...لهم قلوب لا یفقهون بها...﴾.[16]
پس، ندیدن چیزی دلیل روشن نبودن آن شی‏ء نیست؛ زیرا ندیدن چیزی ممكن است در اثر تاریك بودن مرئی باشد یا در اثر كوری رائی؛ اما دیدن چیزی دلیل دو مطلب است: یكی آنكه مرئی روشن است، دیگر آنكه رائی بیناست. خداوند سبحانْ، قرآن را روشن معرفی كرد و فهمندگان آن را بینا و شنوا؛ ﴿...لمن كان له قلب أو ألقی السمع و هو شهید﴾[17] و محجوبان از فهم آن را كور، كر، و دِلْ مكنون و دِلْ مطبوع و دِلْ مختوم و مانند آن یاد نمود.
کتاب مبین بودن قرآن
پنجم. قرآن كریم كه عهده‏دار بیان تمام معارف و احكامی است كه در تأمین هدایت، سعادت و سیادت جوامع انسانی سهم مؤثری دارد، در تبیین ره آورد خود و تفسیر خویش، نباید مُبْهَم و مجمل باشد؛ زیرا كتاب مبهم كه قادرِ حلّ معانی خویشتن و تفسیر مطالب خود نمی‏باشد، هرگز توان تبیین معارف سعادت بخش را نخواهد داشت؛ چنان‏كه خداوند سبحان دربارهٴ قرآن چنین فرموده است: ﴿...و نزّلنا علیك الكتاب تبیاناً لكل شی‏ء...﴾.[18]
از این جهت، قرآن كریم، هم نسبت به محدودهٴ داخلی خودْ بَیّن است و هم نسبت به قلمرو بیرون از خویش مُبَیّن؛ لذا می‏توان آن را كتاب مُبین به معنای روشن و روشنگر دانست؛ چنان‏كه خداوند در موارد فراوان از آن به عنوان كتاب مُبین یاد فرمود؛ مانند آیهٴ 1 از سوره یوسف و آیهٴ 1 از سورهٴ حجر و آیهٴ 1 از سورهٴ نمل و آیهٴ 2 از سورهٴ قصص و آیهٴ 69 از سورهٴ یس و....
گذشته از مطلب مزبور، یعنی صرف نظر از آنكه قرآن نور است و مُبین می‏باشد، خداوند سبحانْ دو بیان دیگر دربارهٴ هماهنگی سراسر قرآن با یك‏دیگر دارد كه یكی ناظر به انسجام وعدم اختلاف آیات قرآن مجید با یك‏دیگر و دیگری راجع به اِنْثِناء، اِنْعِطاف، گرایش آیات قرآن با همدیگر می‏باشد.
امّا بیان اوّلِ خداوند، مستفاد از آیهٴ ﴿أفلا یتدبّرون القران و لو كان من عند غیر اللّه لوجدوا فیه اختلافاً كثیراً﴾[19] می‏باشد؛ زیرا پیام آیهٴ مزبور، دعوت همگان به تدبّر تام در سراسر كتابی است كه در طول ربعِ قرن تقریبی در شرایط صَعْب و سَهْل، جنگ و صلح، غُرْبت و قربت، اوج و حضیض، سَرّاء و ضَرّاء و هزیمت و عزیمت، شكست و پیروزی و بالاخره در احوال مختلف سیاسی، نظامی، اجتماعی به صورت یكسان و هماهنگ نازل شد.
تحلیل این قیاس استثنایی و تقریر تلازم مقدّم و تالی و تقریب بطلان تالی و استنتاج بطلان مقدّم از ابطال تالی، به استمداد عقل برهانی است كه خود از منابع غنیّ و قویِّ تفسیر قرآن كریم از درون دین است؛ نه از بیرون؛ چنان‏كه در فصل تفسیر قرآن به رأی بازگو شد. غرض آنكه آیهٴ مزبور، دعوت به تحتّم تدبّر در سراسر قرآن می‏نماید و دعوای عدم اختلاف به نحو سالبهٴ كلیه را ارائه می‏كند و محصول آن تدبّرِ فراگیر را صحّت چنین دعوی می‏داند و ادعای مزبور را مشفوع به آن بیّنه و گواه صادق قرار می‏دهد. تحلیل عدم اختلاف آیات قرآن با یك‏دیگر در عین اشتمال بر ناسخ و منسوخ، عامّ و خاص، مطلق و مقیّد، مُجمل و مبیّن، به عهدهٴ عقل مسلّح به سلاح برهانِ ناب است كه چنین عقلی از منابع درون دینی محسوب می‏گردد.
امّا بیان دوم خداوند، مستنبط از آیهٴ ﴿اللّه نزّل أحْسَن الحدیثِ كتاباً متشابهاً مَثانِیَ تقشعّر منه جلود الذین یخشون ربّهم ثم تلین جلودهم و قلوبهم إلی ذكر اللّه ذلك هدی اللّه یهدی به من یشاء و مَن یضلل اللّه فما له من هادٍ﴾[20] می‏باشد؛ زیرا محتوای آیه مزبور این است كه سراسر آیات قرآن مجیدْ شبیه، همسان، همتای هم‏اند از یك سو، مُنثنی، مُنعطف، متمایل هم‏اند از سوی دیگر.
معنای إنثناء، انعطاف و گرایش مطالب یك كتاب علمی، به این است كه هركدام از مطالب با دیگریْ مشروح یا مشروح‏تر، روشن یا روشن‏ترمی‏گردند. چنین كتابی كه تمام آیات آن با هم انثناء و انْعِطِاف دارند، حتماً مفسّر یك‏دیگر، مُبیّن همدیگر و شارح داخلی خویش‏اند.
لازم است توجه شود كه معنای متشابه در این آیه به معنای شبیه هم بودن است، ولی در آیهٴ 7 سورهٴ آل‏عمران، از شبهه‏ناك بودن است كه البته وجه تسمیهٴ شبهه به شبهه، این است كه شبیه حق است، ولی حقّ نیست؛ «إنّها سُمّیت الشبهة، شبهةً لأنها تُشبهُ الحقّ...»[21] ؛ البته اشتباه آن نسبت به جاهل است؛ چنان‏كه از امام صادق(علیهم‏السلام) نقل شد كه: «و المتشابه ما اشتبه علی جاهله».[22]
تفسیر قرآن به قرآن، منهج تفسیری اهل بیت
ششم. منهج اهل بیت عصمت و طهارت(علیهم‏السلام) یعنی رسول اكرم(صلی الله علیه و آله و سلم) و ائمّهٴ معصومین(علیهم‏السلام) در تفسیر، همانا تفسیر قرآن به قرآن و ارجاع مفسّران به چنین روش بوده است؛ یعنی هم اصل مطلب كه قرآن را باید با خود قرآن شناخت، از آنان رسیده است و هم اینكه در مقام عمل، آیات قرآن را به كمك همدیگر تفسیر می‏نمودند كه نقل بخش تفسیری و عملی آنان خارج از مبحث كنونی است و فقط به ارجاع علمی و دستوری آن ذوات مقدّس بسنده می‏شود.
حضرت رسول اكرم(صلی الله علیه و آله و سلم) فرموده‏اند: «إنّ القرآن لیصدّق بعضه بعضاً فلا تكذبوا بعضه ببعضٍ»[23] و حضرت علی(علیهم‏السلام) فرموده‏اند: «كتاب اللّه تبصرون به و تنطقون به و تسمعون به و ینطق بعضه ببعض و یشهد بعضه علی بعضٍ».[24]
منظور از تصدیق آیات قرآن نسبت به یك‏دیگر كه در كلام رسول اكرم(صلی الله علیه و آله و سلم) آمده است، تصدیق اصطلاحی نیست كه در قبال تصور باشد؛ بلكه به معنای همان نطق و شهادت آیات نسبت به یك‏دیگر است كه در كلام حضرت علی(علیهم‏السلام) آمده است، یعنی اگر مبدأ تصوری یك آیه در بین معانی محتمل آن توسط معنای تصوری آیهٴ دیگر كه واضح استْ تفسیر شود، از سنخ تصدیق برخی از آیات نسبت به بعض دیگر است.
زیرا شهادت و همچنین نطق مزبور در این باره نیز صادق است؛ چنان‏كه اگر مبدأ تصدیقی یك آیه در بین معانی و مقاصد متعدد از یك جملهٴ قرآنی توسط جملهٴ دیگر كه مقصود آشكاری دارد (به اصطلاح یكی ظاهر است و دیگری اظهر یا یكی ظاهر است و دیگری نصّ و یا اینكه یكی ظهور مشترك دارد و دیگری ظهور خاص) حلّ گردد، چنین نطق و شهادتی مصداق تصدیق مأخوذ در كلام رسول‏اكرم(صلی الله علیه و آله و سلم) بوده و به نوبهٴ خود، تفسیر قرآن به قرآن محسوب است.
بنابراین، هرگونه شهادت و نطقی كه نسبت به معنای تصوری یا تصدیقی آیه، توسط آیهٴ دیگر حاصل شود، از قبیل تفسیر قرآن به قرآن است و هرگز تصدیق و شهادت، اختصاصی به بعد از استقرار ظهور آیه نخواهد داشت؛ چنان‏كه بعضی گمان كرده‏اند؛[25] چون از طرف اهل‏بیت وحی و عصمت و طهارت(علیهم‏السلام) یعنی رسول‏اكرم(صلی الله علیه و آله و سلم) و ائمّهٴ معصومین(علیهم‏السلام) تفسیر قرآن به قرآن به عنوان یك منهج معقول و مقبول اعلام شد.
صحابه و تابعان آنها اگر شاهدی از قرآن برای تفسیر آیه‏ای از آن می‏یافتند، به تفسیر آن مبادرت می‏نمودند؛ گرچه غالب تفسیرهای آنان از سنخ تفسیر به مأثور بود؛ نه تفسیر قرآن به قرآن كه چنین كاری اجتهاد قرآنی و تدبّر در محور تمام آیات را می‏طلبید و نه تفسیر درایی محمود و ممدوح كه اجتهاد برهانی و تأمّل در مدار علوم متعارفه یا اصول موضوعه را كه در جای خود مُبرَهْنْ شده، طلب می‏نمود.
لیكن از دیر زمان، جریان تفسیر قرآن به قرآن بین قرآن پژوهان مطرح بود. راقم سطور، هم در تفسیر ابو جعفر محمد بن جریر طبری (م 310) و هم در تفسیر فخر رازی جملهٴ «القرآن یفسّر بعضه بعضاً» را دیده است و در حال نگارش این مقدمه و شرایط مخصوص آن ، امكان فحص مجدد را ندارم و روش عملی بسیاری از بزرگان تفسیر نیز استمداد از بعض آیات دربارهٴ بعض دیگر، «فی الجمله» نه «بالجمله» بوده است.
مؤلف تفسیر المنار، شرایطی را برای مرحلهٴ عالیهٴ از تفسیر یاد می‏نماید كه اوّلین آن شرایط و امورِ لازم، بررسی خود آیات و كلمات برای فهم خود قرآن است. آنگاه چنین گفته است: «و قد قالوا إنّ القرآن یفسّر بعضه ببعضٍ»[26]. این تعبیر نشان می‏دهد كه جریان تفسیر قرآن به قرآن، قولی است كه جملگی بر آن بوده‏اند؛ نه اینكه مختص به گروهی معیّن باشد؛ اما نه خود صاحب المنار، جناب محمد عبده و رشید رضا و نه دیگران توان چنین كار سترگی را نداشتند كه قرآن كریم را به نوبهٴ خویش استنطاق نمایند و آن را به نطق آورند تا آیات همگون آن نسبت به همدیگر ناطق، شاهد و مصدّق باشند؛ گرچه فی الجمله در این منهج بهیج توفیق یافتند. غرض آنكه از گفتار و رفتار و نوشتار اَقْدَمین، قدما، متأخران و معاصران، عطرِ دل‏انگیزِ تفسیر قرآن به قرآن به شامّهٴ جان می‏وزد.
لیكن چنین نافه‏ای را باید در سوق كالای گرانبهای المیزان جست‏وجو نمود كه در این مضمار، گوی سَبَق را از دیگران ربود و هرچند از لحاظ شناسنامهٴ تفسیری، آرم بنوّت و فرزندی مفسّران سَلَف را داراست، لكن «فیه معنیً شاهد بأُبوّته».[27]
اگرچه حُسن فروشان به جلوه آمده‏اند ٭٭٭٭ كسی به حُسن و ملاحت به یار ما نرسد
به حق صحبت دیرین كه هیچ محرم راز ٭٭٭٭ به یار یك جهت حق گزار ما نرسد

هزار نقد به بازار كائنات آرند ٭٭٭٭ یكی به سكهٴ صاحب عیار ما نرسد[28]
المیزان و ما ادراك ما المیزان، ﴿ذلك فضل اللّه یؤتیه من یشاء﴾[29]. راز كامیابی مؤلف بزرگوار المیزان، در تفسیر قرآن به قرآن بعداً روشن خواهد شد.
آسان یا دشواری تفسیرقرآن به قرآن درگرومیزان بهره مندی مفسرازعلوم حصولی وحضوری
هفتم. تفسیر قرآن به قرآن، انحاء گونه گون دارد كه برخی از آن صُوَرْ سَهْل و بعضی از آنها صَعْب می‏باشد. بعضی از اقسام آن به قدری سهل است كه شاید عنوان تفسیر قرآن به قرآن از آن منصرف باشد؛ چنان‏كه برخی به قدری پیچیده است كه به دشواری به ذهن مفسّر فحّاصِ متّتبع می‏آید و چون پیوند آیات مورد تفسیر همیشه از سنخ واژه‏ها نیست، شاید برخی آن را از سنخ تفسیر قرآن به قرآن ندانند. به هر تقدیر، عنوان تفسیر قرآن به قرآن، نه دارای مرزبندی لغوی است و نه دارای حقیقت شرعیّه؛ بلكه وابسته به عمق غَوْص مفسّر دریانورد یا به سَمْك و اوج مفسّر آسمان سا و فضا نورد می‏باشد؛ تا چه اندازه با علم حصولی پرواز كند یا چه مقدار به علم حضوری غوّاصی نماید.
