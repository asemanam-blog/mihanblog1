بگو که گل نفرستد کسی به خانه ی من ...

بگو که گل نفرستد کســـــی به خانه من
که عطر یاد تو پر کرده آشیـــــــــــــانه من
تو چلــــــــــچراغ سعادت فروز بخت منـی
بجای مـــــــــاه تو پرتو فشان به خـانه من
به شوق روی تو من زنــــــــده ام خدا داند
برای زیستن اینــــــــــــــک تویی بهانه من

صدام کن ای صداقت پیشه، بی بی گـــــــل عتیقه
تمام دلخوشــــــــــــــــــــــیم اینه که دل با تو رفیقه
صدام کن ای هــــــــــــــــــــوای تازه ای عطر رمنده
هوا پر شه پر از پــــــــــــــــــــــــــرهای رنگی پرنده
بزن بارون بزن خیســــــــــــــم کن، آبم کن، ترم کن
کمک کن تازه بارون من غـــــــــــــــــــریبم پرپرم کن
بزن آتیش به جونم شعله کن خاکســــــــــــترم کن
بذار سر روی دوشم سایه ت رو تاج ســـــــــرم کن
صدام کن ای صداقـــــــت پیشه، بی بی گل عتیقه
تمام دلخوشیم اینــــــــــــــــــــــه که دل با تو رفیقه

تو عاشق پیشه ای همیشه ای محشربپا کن

 منو عاشــــــــــــــــق ترین آواره عالم صدا کن  

  من  از مکتب سراغ قـبله عــــــــشقو گرفتم                                  

تو اینجا تا ابد از عشــــــــــــق مردن رو بنا کن

با سقوط دستای ما

با سقــــــــــوط دســـــــــــــتای ما
در تنم چیزی فــــــــــــــــــرو ریخت
هجرتـــــــــــــــــــــــت اوج صدامو
از فراز شـــــــــــــــــــــاخه آویـخت
ای زلال ســـــــــــــبز جــــــــــاری
جای خـــــــــــوب غســــــل تعمید
بی تو باید مــــــــــــــــــرد و پژمـرد
زیر خاک باغچه پوســـــــــــــــــــید

فصــــــــــــــــــــــــلی که مـــــــــن با تو ماشــــــــد
فصــــــــــــــــــــــــــــــل ســــــــــــبز خواهش برگ
فصلـــــــــــــــــــــــــــی که ما بی تو مــــــــــن شد
فصــــــــــــــــــــــــــــــــــل خاکســـــــــــتری مرگ

تو بگو جز تو کـــــــــدوم رود
ناجی لب تشـــــــــنگی بود
جز تو آغـــــــــوش کدوم باد
سایه گاه خســــــــتگی بود
بــــــی تو بایـــــد بی تو باید
تانفـــــــــــــــس دارم ببارم
من برای گــــــــــــریه کردن
شونه هــــــاتو کم می یارم

چشــــــــــــــــم تو با هق هق مــــن
با شکـــــــــستن آشنا نیـــــــــــست
این شکــــــــــــــــــستن بی صدا بود
هر صـــــــــــــــــدایی که صدا نیست

ای رفــــــــــــیق ناخوشی ها
این خوشــــــــــــی باید بمیره
جز تو همراهــــــــــــی ندارم
تا شب از من پس بگـــــــــیره
با تو بدرود ای مـســــــــــــافر
هجــــــــــــرت تو بی خطر باد
پر تپش باشــــــــــــه دلی که
خون به رگ های تنــــــــم داد

سر ساقی سلامت

سلام من بــــــــه تو یــــــــار قدیمــــی ،منم همــــون هـــــوادار قدیمــــــی

هنـــــــوز همـــــون خرابـــــــــاتی و مستم ،ولـی بــــی تو سبوی می شکستم

همه تشنه لبیم ســـــــــــــــاقی کجایـــــی؟ گرفتـــار شبیم ساقی کجــــــــــــــایی؟

اگه سبــــــــــو شکست،عمـــــر تو باقـــی که اعتبار می،تویی تو ســــــــــــــــــــــــاقی!

اگه میکده امروز شده خونـــــــــــه تزویــر،تو محراب دل ما،تویــــــی تو مرشد و پیــــــــــــر!

همه به جرم مستی،سر دار ملامت،میمیریم و میخونیــــــــــم:سر ساقـــــــــی سلامت

یه روزی گلـــــــه کردم من از عالم مستیتو هم به دل گرفتی دل ما رو شکستــــــــی

من از مستی نوشتم،ولی قلب تو رنجید،تو قهر کردی و قهرت مصیبت شد و بارید

پشیمونـــــم و خستـــــــــــم، اگـــــــــــــــــه عهدی شکستــــــــــــــــــم

آخــــــــــــه مست تو هستم ،اگـــــــه مجرم و مستـــــــــم

 همـــــــــــه به جرم مستی ســـر دار ملامت

فرق من و تو اینه

خوب به خودت نگاه کن
ببین چـــی داری از من
جز اینکه صد تا دنیـــــــا
فاصله داری از مـــــــــن
فاصله امون یه دنیاست
پس انتـــــــــــــــها نداره
ما مال هم نمی شــیم
تو روز و من ســـــــــتاره

نه قلب تو بند منه، نه فکرمی همیــــــــشه
بیا به هم دروغ نگیم، اینجوری حل نمـیشه
نه یاد من مونده نه تو، که قلب کی نحیــفه
مسلما بیشتر از این، حافظه امون ضعــیفه

مهم نبود از اول که آخــــــــــرش چی میشه
نگفــــــــــــــــــــته ها زیاده، اما این آخریشه
هرکی بخواد می تونه، تو قلـــــب تو بشینه
خوب به خودت نگــاه کن، فرق من و تو اینه
نه قلب تو بند منه، نه فکــــــــرمی همیشه
بیا به هم دروغ نگیم، اینجـوری حل نمیشه
